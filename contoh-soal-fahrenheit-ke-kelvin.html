Penjelasan:

30° C

R = 4/5 C

= 4/5 (30)

= 24° R

F = 9/5 C + 32

= 9/5 (30) + 32

= 54 + 32

= 86° F

K = C + 273

= 30 + 273

= 303 K

20° R

C = 5/4 R

= 5/4 (20)

= 25° C

F = 9/4 R + 32

= 9/4 (20) + 32

= 45 + 32

= 77° F

K = 5/4 R + 273

= 5/4 (20) + 273

= 25 + 273

= 298 K

68° F

C = 5/9 (F - 32)

= 5/9 (68 - 32)

= 5/9 (36)

= 20° C

R = 4/9 (F - 32)

= 4/9 (36)

= 16° R

K = 5/9 (F - 32) + 273

= 5/9 (36) + 273

= 20 + 273

= 293 K

293 K

C = K - 273

= 293 - 273

= 20° C

R = 4/5 (K - 273)

= 4/5 (20)

= 16° R

F = 9/5 (K - 273) + 32

= 9/5 (20) + 32

= 36 + 32

= 68° F
