#include <iostream>

using namespace std;

char papan[3][3] = {{'1', '2', '3'}, {'4', '5', '6'}, {'7', '8', '9'}};

char giliran = 'X';

bool selesai = false;

void TampilkanPapan() {

system("cls"); // Menghapus layar untuk tampilan yang lebih baik

cout << "Tic Tac Toe\n\n";

for (int i = 0; i < 3; i++) {

for (int j = 0; j < 3; j++) {

cout << papan[i][j];

if (j < 2) {

cout << " | ";

}

}

cout << endl;

if (i < 2) {

cout << "---------\n";

}

}

cout << endl;

}

void GantiGiliran() {

if (giliran == 'X') {

giliran = 'O';

} else {

giliran = 'X';

}

}

bool LakukanLangkah(int langkah) {

for (int i = 0; i < 3; i++) {

for (int j = 0; j < 3; j++) {

if (papan[i][j] == '0' + langkah) {

if (papan[i][j] == 'X' || papan[i][j] == 'O') {

return false; // Langkah sudah diambil

}

papan[i][j] = giliran;

return true;

}

}

}

return false; // Langkah tidak valid

}

bool CekKemenangan() {

for (int i = 0; i < 3; i++) {

if (papan[i][0] == papan[i][1] && papan[i][1] == papan[i][2]) {

return true; // Pemenang horizontal

}

}

for (int i = 0; i < 3; i++) {

if (papan[0][i] == papan[1][i] && papan[1][i] == papan[2][i]) {

return true; // Pemenang vertikal

}

}

if (papan[0][0] == papan[1][1] && papan[1][1] == papan[2][2]) {

return true; // Pemenang diagonal (kiri atas - kanan bawah)

}

if (papan[0][2] == papan[1][1] && papan[1][1] == papan[2][0]) {

return true; // Pemenang diagonal (kanan atas - kiri bawah)

}

return false;

}

int main() {

int langkah;

TampilkanPapan();

while (!selesai) {

cout << "Giliran pemain " << giliran << ". Masukkan langkah (1-9): ";

cin >> langkah;

if (langkah < 1 || langkah > 9) {

cout << "Langkah tidak valid. Coba lagi.\n";

continue;

}

if (!LakukanLangkah(langkah)) {

cout << "Langkah sudah diambil. Coba lagi.\n";

continue;

}

TampilkanPapan();

if (CekKemenangan()) {

cout << "Pemain " << giliran << " memenangkan permainan!\n";

selesai = true;

} else if (papan[0][0] != '1' && papan[0][1] != '2' && papan[0][2] != '3' &&

papan[1][0] != '4' && papan[1][1] != '5' && papan[1][2] != '6' &&

papan[2][0] != '7' && papan[2][1] != '8' && papan[2][2] != '9') {

cout << "Permainan berakhir dengan hasil seri!\n";

selesai = true;

} else {

GantiGiliran();

}

}

return 0;

}

18. Sejarahnya terbentuknya TAC


Tanjungpinang Aeromodelling Club atau yang lebih dikenal dengan TAC sebenarnya sudah ada sejak era tahun 80an, hal tersebut terbukti dari peran serta TAC di stand Pameran Pembangunan milik TNI AU. Menurut informasi dari para sesepuh sesepuh pendiri TAC, mereka selalu giat berlatih dan berkumpul di Bandar Udara Kijang (sekarang Raja Haji Fisabilillah) dan memiliki banyak member bahkan selalu terkoneksi dengan aeromodeller di negara tetangga Singapura, yang selalu bergabung bermain bersama di Bandara Kijang. Hanya saja pada saat itu organisasi/ club yang bernama TAC ini hanya sebatas komunitas hobby saja tanpa ada legalitas pendukungnya.

Baru sejak tahun 2009 dengan semakin maraknya olah raga aeromodelling di Tanjungpinang, para sesepuh bersama generasi baru TAC berkumpul bersama menggagas baik itu anggaran dasar, anggaran rumah tangga maupun legalitas lainnya guna mendukung eksistensi TAC kedepannya. Maka sejak akhir tahun 2009 dengan didahului oleh rapat rapat sebelumnya maka tepatnya tanggal 15 Nopember 2009 pembentukan pengurus dan organisasi resmi diputuskan. Sejak saat itu ditetapkanlah tanggal tersebut sebagai tanggal lahirnya Tanjungpinang Aeromodelling Club (TAC).
Pada hari itu rapat dihadiri oleh beberapa orang sesepuh dan generasi baru TAC antara lain :

1. Mayor Laut Elfanda Syarif
2. Hari Maxi
3. Doddy Oliver
4. Danny Dalco
5. Satrio Aribowo
6. Yanuar
7. Irwan
8. Mulyadi
9. Hendri
10. Alex Lie

Orang orang tersebut lah yang membidani lahirnya TAC baik secara defacto maupun dejure. Hasil rapat tersebut memutuskan hal - hal antara lain menetapkan draft AD/ART TAC, merencanakan pemilihan Ketua TAC, menyusun draft kepengurusan TAC tahun 2009-2013.

Akhirnya pada tanggal 11 Januari 2010 terbentuklah kepengurusan TAC yang baru, yang dipimpin oleh Ketua TAC pertama yaitu Sdr. Bowo Moko dengan tugas tugas utama antara lain adalah segera melegalisasi oganisasi dimaksud melalu notaris dan mendaftarkan ke Kesbangpol Linmas Tanjungpinang, Lalu melaporkannya ke Komandan Lanud Tanjungpinang selaku Ketua Federasi Aerosport Indonesia Kepulauan Riau dan meminta kesediaan beliau agar TAC dapat segera dikukuhkan.

Akhirnya TAC menjadi organisasi aeromodelling yang resmi dan juga telah didaftarkan ke PB FASI menjadi salah satu wadah bagi para aeromodellers untuk berbagi informasi, pengetahuan, teknologi serta mencari bibit-bibit baru atlit-atlit aeromodelling berprestasi.

Hal ini terjawab dengan prestasi-prestasi atlit TAC di pertandingan baik tingkat nasional maupun internasional, antara lain Juara 2 Kejuaraan Pylon Race terbuka di Thailand tahun 2009, Medali Emas dan perunggu di Kejuaraan Nasional Aeromodelling di Musi Rawas (Sumatera Selatan) Tahun 2010, Medali Perunggu PON XVIII 2012 Riau, Medali Perak dan Perunggu di PON XIX 2016 Jawa Barat.

Tentu saja TAC tidak hanya berhenti sampai di sini, dengan program kerja yang dimiliki TAC diharapkan TAC menjadi salah satu sumber atlit nasional yang ikut bertanding di Kejuaraan Dunia Aeromodelling, dan menjadi salah satu barometer aeromodelling di Indonesia.

Mekanisme kerja TAC dalam konflik antar negara ialah dengan melibatkan pihak ketiga dalam menyelesaikan konflik sehingga konflik dapat terselesaikan dengan damai tanpa pertikaian atau perseteruan. Pihak ketiga bertugas sebagai hakim apabila terjadi konflik, maka dari itu pihak ketiga yang dipilih harus bersifat netral (tidak memihak pada salah satu pihak.  

Penjelasan:

TAC (Treaty Amity and Cooperation) merupakan suatu perjanjian persahataban dan kerjasama multinasional dalam lingkup ASEAN dengan melibatkan pihak ketiga diluar anggota ASEAN apabila terjadi konflik. TAC disetujui pada tanggal 24 Februari 1976 di Denpasar, Bali, Indonesia. Perjanjian ini awalnya disetujui dan ditandatangani oleh Indonesia, Malaysia, Singapura, Thailand, dan Filipina. Lalu, secara berturut-turut diikuti dengan Brunei, Papua Nugini, Laos, Vietnam, Kamboja, Myanmar, China, India, Jepang, Pakistan, Korea Selatan, Rusia, Selandia Baru, Mongolia, Australia, Prancis, Timor Leste, Bangladesh, Srilanka, Korea Utara, Inggris, Amerika Serikat dan terakhir Brazil.

TAC dibentuk sebagai alat untuk menyelesaikan masalah dan pertentangan dalam lingkup Asia Tenggara. Hal ini dilakukan untuk menghindari penggunaan ancaman atau kekerasan dalam menyelesaikan konflik yang terjadi dalam lingkup Asia Tenggara.

34. Sebutkan 6 Tac ASEAN​


Jawaban:

Berikut adalah enam prinsip atau tac ASEAN yang terkandung dalam TAC:

1.Saling Menghormati Kedaulatan dan Integritas Wilayah: Negara-negara anggota ASEAN berjanji untuk menghormati kedaulatan dan integritas wilayah masing-masing serta tidak melakukan campur tangan dalam urusan dalam negeri negara-negara lain.

2.Tidak Mengancam atau Menggunakan Kekuatan: Negara-negara anggota sepakat untuk tidak mengancam atau menggunakan kekuatan dalam penyelesaian sengketa dan konflik regional.

3.Prinsip Non-Interferensi: ASEAN menganut prinsip non-interferensi, yang berarti negara-negara anggota tidak boleh campur tangan dalam urusan dalam negeri negara-negara lain.

4.Solusi Damai dan Diplomasi: Negara-negara anggota ASEAN berkomitmen untuk menyelesaikan sengketa dan konflik melalui jalur diplomatik dan negosiasi, bukan dengan kekerasan.

5.Penghormatan terhadap Hak Asasi Manusia: Prinsip TAC ASEAN juga mencakup penghormatan terhadap hak asasi manusia dan prinsip-prinsip demokrasi.

6.Kerjasama dan Persahabatan: TAC ASEAN menekankan pentingnya kerjasama yang erat dan persahabatan antara negara-negara anggota untuk mencapai perdamaian dan kestabilan di kawasan Asia Tenggara.
