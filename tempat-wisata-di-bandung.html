himpunan siswa minat ke tempat wisata = A ∩ B

|A| = 29 siswa milih bandung

|B| = 32 milih bali

A ∩ B = |A| + |B| - |A ∪ B|

A ∪ B:

A ∪ B = |A| + |B| - |A ∩ B|

= 29 + 32 - |A ∩ B|

= 61 - |A ∩ B|

A ∩ B = |A| + |B| - |A ∪ B|

= 29 + 32 - (61 - |A ∩ B|)

= 61 - 29 - 32 + |A ∩ B|

= |A ∩ B|

= 29 + 32 - 61 = 0

B - A = |B| - |A ∩ B| (siswa ke bali)

A ∩ B = 0

B - A = |B|
