17. perbandingan permen hera dan hero 5:3jika selisih permen mereka 12hitunglah jumlah permen hera dan hero​


Jawaban:

Permen Hera

[tex] = \frac{5}{5 - 3} \times 12 \\ = \frac{5}{2} \times 12 \\ = 5 \times 6 \\ = 30[/tex]

Permen Hero

[tex] = \frac{3}{5 - 3} \times 12 \\ = \frac{3}{2} \times 12 \\ = 3 \times 6 \\ = 18[/tex]

Jadi jumlah permen hera 30 dan permen hero 18

19. 4. Buatlah program yang dapat menerima input dari user kemudian menghasilkan bilangan random . Apabila bilangan random yang dihasilkan adalah bilangan ganjil, maka bilangan tersebut ditambah dengan 1 Contoh Tampilan course hero


import java.util.Scanner;

import java.util.Random;

class RandomNumber{

public static void main(String[] args){

 RandomNumber randomNumber = new RandomNumber();

 randomNumber.inputArray();

 System.out.println("\n\n~sen");

}

 

private void inputArray(){

 Scanner input = new Scanner(System.in);

 System.out.print("Jumlah angka yang akan diacak: ");

 int count = input.nextInt();

 

 int[] numbers = new int[count];

 int i = 0;

 

 while(i < count){

  numbers[i] = input.nextInt();

  i++;

 }

 findRandom(numbers, count);

}

 

private void findRandom(int numbers[], int count){

 Random random = new Random();

 boolean notSame = true;

 

 do{

 int number = random.nextInt(max(numbers, count));

 

  for(int i = 0; i < count; i++){

   if(numbers[i] == number){

    if(number % 2 == 1)

     System.out.printf("\n%d(+1) = %d", number, number + 1);

    else if(number % 2 == 0)

     System.out.printf("\n%d", number);

     

    notSame = false;

    break;

   }

   else if(i == count)

    notSame = true;

  }

 }while(notSame == true);

}

 

private int max(int numbers[], int count){

 int max = numbers[0];

 for(int i = 0; i < count - 1; i += 2){

  if(i + 1 > count)

   if(numbers[i] > max) max = numbers[i];

   

  if(numbers[i] >= numbers[i + 1])

               if(numbers[i] > max) max = numbers[i];

           

           if(numbers[i] <= numbers[i + 1])  

               if(numbers[i + 1] > max) max = numbers[i + 1];

 }

 return max;

}

}
