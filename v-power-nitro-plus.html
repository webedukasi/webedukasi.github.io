              Δt' = Δt / √(1 - (v₁/c)²)
          ⁵/₄ T₀ = T₀ / √(1 - (v₁/c)²)
               ⁵/₄ = 1 / √(1 - (v₁/c)²)
√(1 - (v₁/c)²) = ⁴/₅
     1 - (v₁/c)² = ¹⁶/₂₅
          (v₁/c)² = ⁹/₂₅
              v₁/c = ³/₅
                 v₁ = ³/₅ c

              Δt' = Δt / √(1 - (v₂/c)²)
          ⁵/₃ T₀ = T₀ / √(1 - (v₂/c)²)
               ⁵/₃ = 1 / √(1 - (v₂/c)²)
√(1 - (v₂/c)²) = ³/₅
     1 - (v₂/c)² = ⁹/₂₅
          (v₂/c)² = ¹⁶/₂₅
              v₂/c = ⁴/₅
                 v₂ = ⁴/₅ c

Jadi
v₁ / v₂ = ³/₅ c : ⁴/₅ c
